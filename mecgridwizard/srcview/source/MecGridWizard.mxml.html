        public function DecriptData(source:String):String
        {
            var decsc:String = "";
            
            try 
            {
                var bt:ByteArray;

                var dec:Base64Decoder = new Base64Decoder();
                dec.decode(source);
                bt = dec.flush();
                bt.uncompress();
                bt.position = 0;
                decsc = bt.readUTF();
            }
            catch (e:Error)
            {
                trace("DecriptData Error");
            }
            return decsc;
        }
        
        public function EncriptData(source:String):String
        {
            var scenc:String = "";
            try
            {            
                var enc:Base64Encoder = new Base64Encoder();
                var sc:String = source;
                var bt:ByteArray = new ByteArray();
                bt.writeUTF(sc);
                bt.compress();
                enc.encodeBytes(bt, 0);
                
                scenc = enc.flush();
            }
            catch (e:Error)
            {
                trace("EncriptData Error");
            }
            
            return scenc;
        }

        public function treeChanged(event:Event):void 
        {
            selectedNode=Tree(event.target).selectedItem as XML;
            
            if (selectedNode.hasSimpleContent() == true && selectedNode.@columngroup.toString() == "" && selectedNode.@stylegroup.toString() == "")
            {
                if (selectedNode.@column.toString() != "")
                {
                    if (selectedColumnIndex > -1 && vprop.selectedChild == columnoptions)
                    {
                        applyUpdateProperty();
                        // validateColumnProperty(selectedColumnIndex);
                    }
                    
                    selectedColumnIndex = Number(selectedNode.@column.toString());
                    vprop.selectedChild = columnoptions;
                    
                    validateColumnProperty(selectedColumnIndex);
                }
                else if (selectedNode.@style.toString() != "")
                {
                    if (selectedStyleName != null && vprop.selectedChild == styleoptions)
                    {
                        applyUpdateProperty();
                    }
                    
                    selectedStyleName = selectedNode.@style.toString();
                    vprop.selectedChild = styleoptions;
                    
                    validateStyleProperty(selectedStyleName);
                }
                else
                {
                    var sid:String = selectedNode.@id.toString();
                    vprop.selectedIndex = Number(sid);
                    
                    if (vprop.selectedChild == columndesign)
                    {
                        this.drawColumnWizard();
                    }
                }
            }
        }

        private function inverseApplyProperty():void
        {
            var i:int;
            
            this.selectedColumnIndex = -1;
            this.optCols.value = mgrid.Cols.count;
            this.optFixedCol.value = mgrid.Cols.fixed;
            this.optFixedRow.value = mgrid.Rows.fixed;
            
            this.optExtendLastColumn.selected = mgrid.ExtendLastColumn;
            
            this.spreadsheetindex.selected = mgrid.spreadSheetStyle;
            
            if (mgrid.merge_option & MecGridConst.MERGE_NONE)
            {
                setMergeOption(this.merge_normal, MecGridConst.MERGE_NONE);
            }
            else if (mgrid.merge_option & MecGridConst.MERGE_PREV)
            {
                setMergeOption(this.merge_normal, MecGridConst.MERGE_PREV);
            }
            else if (mgrid.merge_option & MecGridConst.MERGE_FREE)
            {
                setMergeOption(this.merge_normal, MecGridConst.MERGE_FREE);
            }
            
            this.normal_fixed.selected = (mgrid.merge_option & MecGridConst.MERGE_FIXED) ? true : false;
            
            setMergeOption(this.merge_fixcol, mgrid.merge_option_fixedcolumn);
            setMergeOption(this.merge_fixrow, mgrid.merge_option_fixedrow);

            this.selRow.selected = (mgrid.SelectionMode & CellOp.SELECTIONMODE_ROW) ? true : false;        
            this.selCol.selected = (mgrid.SelectionMode & CellOp.SELECTIONMODE_CELL) ? true : false;
            
            this.selSingle.selected = (mgrid.SelectionMode & CellOp.SELECT_SINGLE) ? true : false;
            this.selMulti.selected = (mgrid.SelectionMode & CellOp.SELECT_MULTI) ? true : false;
            
            switch (mgrid.SortOption)
            {
            case MecGridConst.SORT_MULTI_COLUMN:
                sortoption.selectedValue = "SORT_MULTI_COLUMN";
                break;
            case MecGridConst.SORT_SINGLE_COLUMN:
                sortoption.selectedValue = "SORT_SINGLE_COLUMN";
                break;
            default:
                sortoption.selectedValue = "SORT_NONE";
                break;
            }
                        
            vprop.selectedIndex = 0;
            
            updateTreeNode();
            
            for (i=0; i < this.textalignment.length; i++)
            {
                if (this.textalignment[i].data == mgrid.TextAlignment)
                {
                    this.default_alignment.selectedIndex = i;
                }
                
                if (this.textalignment[i].data == mgrid.TextAlignmentFixed)
                {
                    this.default_fixed_alignment.selectedIndex = i;
                }
            }
        }
        
        private function applyUpdateProperty():void
        {
            mgrid.redraw = false;
            
            var i:int;
            
            if (vprop.selectedChild == columnoptions)
            {
                trace("column update");
                // selectedColumnIndex
                var c:int = selectedColumnIndex;
                
                if (c < mgrid.Cols.count && c > -1)
                {
                    if (col_name.text == "")
                    {
                        col_name.text = "col_" + c;
                    }
                    mgrid.Cols.item(c).name = col_name.text;
                    mgrid.Cols.item(c).editable = col_editable.selected;
                    mgrid.Cols.item(c).visible = col_visible.selected;
                    mgrid.Cols.item(c).dataField = col_datafield.text;
                    mgrid.Cols.item(c).autowidth = col_autowidth.selected;
                    
                    mgrid.Cols.item(c).datatype = (col_datatype.selectedItem) ? col_datatype.selectedItem.data : CellOp.TEXT;
                    mgrid.Cols.item(c).textalign = (col_textalign.selectedItem) ? col_textalign.selectedItem.data : CellOp.SYS_TEXTALIGN_DEFAULT;
                    
                    mgrid.Cols.item(c).textalign_fixed = (col_textalign_fixed.selectedItem) ? col_textalign_fixed.selectedItem.data : CellOp.SYS_TEXTALIGN_DEFAULT;
                    mgrid.Cols.item(c).merge = col_merge.selected;
                    mgrid.Cols.item(c).enableFilter = this.col_showfilter.selected;
                    
                    if (col_style.selectedItem)
                    {
                        mgrid.Cols.item(c).style = col_style.selectedItem.data;
                    }
                    
                    if (col_autowidth.selected == false)
                        mgrid.Cols.item(c).width = this.col_width.value;
                    
                    var header:Array = col_header.text.split(";");
                    var lheader:String = header[0];
                    
                    for (i=0; i < mgrid.Rows.fixed; i++)
                    {
                        if (header.length > i)
                        {
                            mgrid.set_cell(CellOp.TEXT, i, c, header[i]);
                            lheader = header[i];
                        }
                        else
                        {
                            mgrid.set_cell(CellOp.TEXT, i, c, lheader);
                        }
                    }
                }
            }
            else if (vprop.selectedChild == this.columndesign)
            {
                this.applyColumnWizard();
            }
            else if (vprop.selectedChild == styleoptions)
            {
                this.applyStyleProperty();
            }
            else
            {
                trace("master update");
                
                mgrid.Cols.count = optCols.value;
                mgrid.Rows.count = optFixedRow.value;
                mgrid.Cols.fixed = optFixedCol.value;
                mgrid.Rows.fixed = optFixedRow.value;
                
                mgrid.ExtendLastColumn = optExtendLastColumn.selected;
                
                // mgrid.merge_option = (merge_normal.selectedItem) ? this.merge_normal.selectedItem.data : this.mergeoptions[0].data;
                
                
                mgrid.merge_option = (this.merge_normal.selectedItem) ? this.merge_normal.selectedItem.data : this.mergeoptions[0].data;;
                mgrid.merge_option = (this.normal_fixed.selected) ? mgrid.merge_option | MecGridConst.MERGE_FIXED : mgrid.merge_option;
                
                mgrid.merge_option_fixedcolumn = (this.merge_fixcol.selectedItem) ? this.merge_fixcol.selectedItem.data : this.mergeoptions[0].data;
                mgrid.merge_option_fixedrow = (this.merge_fixrow.selectedItem) ? this.merge_fixrow.selectedItem.data : this.mergeoptions[0].data;
                
                mgrid.spreadSheetStyle = spreadsheetindex.selected;
                
                mgrid.SelectionMode = CellOp.SELECTIONMODE_NONE;
                mgrid.SelectionMode = (this.selRow.selected) ? mgrid.SelectionMode | CellOp.SELECTIONMODE_ROW : mgrid.SelectionMode;
                mgrid.SelectionMode = (this.selCol.selected) ? mgrid.SelectionMode | CellOp.SELECTIONMODE_CELL : mgrid.SelectionMode;
                mgrid.SelectionMode = (this.selSingle.selected) ? mgrid.SelectionMode | CellOp.SELECT_SINGLE : mgrid.SelectionMode;
                mgrid.SelectionMode = (this.selMulti.selected) ? mgrid.SelectionMode | CellOp.SELECT_MULTI : mgrid.SelectionMode;
                
                switch (this.sortoption.selectedValue)
                {
                case "SORT_SINGLE_COLUMN":
                    mgrid.SortOption = MecGridConst.SORT_SINGLE_COLUMN;
                    break;
                case "SORT_MULTI_COLUMN":
                    mgrid.SortOption = MecGridConst.SORT_MULTI_COLUMN;
                    break;
                default:
                    mgrid.SortOption = MecGridConst.SORT_NONE;
                    break;
                }
                
                mgrid.TextAlignment = (this.default_alignment.selectedItem) ? this.default_alignment.selectedItem.data : mgrid.TextAlignment;
                mgrid.TextAlignmentFixed = (this.default_fixed_alignment.selectedItem) ? this.default_fixed_alignment.selectedItem.data : mgrid.TextAlignmentFixed; 
                
                loadSampleData();
                
                updateTreeNode();
            }
            
            mgrid.redraw = true;
        }
        
        private function loadSampleData():void
        {
            var i:int;
            var j:int;
            
            mgrid.Rows.count = mgrid.Rows.fixed + 10;
            
            for (i=mgrid.Rows.fixed; i < mgrid.Rows.count; i++)
            {
                for (j=0; j < mgrid.Cols.count; j++)
                {
                    switch (mgrid.Cols.item(j).datatype)
                    {
                    case CellOp.TEXT:
                    case CellOp.STATUSBAR:
                        mgrid.set_cell(CellOp.TEXT, i, j, (Math.random() * 100).toFixed(5));
                        break;
                    case CellOp.BOOLEAN:
                        mgrid.set_cell(CellOp.BOOLEAN, i, j, ((Math.random() > 0.5) ? true : false));
                        break;
                    }
                }
            } 
        }
        
        private function updateTreeNode():void
        {
            var i:int = 0;
            var columninfo:XML;
            var styleinfo:XML;
            
            for (i=0; i < treeNavData.length(); i++)
            {
                if (treeNavData[i].@columngroup.toString() == "T")
                {
                    columninfo = treeNavData[i];
                    break;
                }
            }
            
            if (columninfo.children().length() != mgrid.Cols.count)
            {
                if (columninfo.hasComplexContent())
                {
                    while (!columninfo.hasSimpleContent())
                        delete columninfo.children()[0];
                }
                    
                for (i=0; i < mgrid.Cols.count; i++)
                {
                    var m_child:XML = new XML(<node/>);
                    m_child.@column = i;
                    m_child.@label = "col_" + i;
                    columninfo.appendChild(m_child);
                }
            }
            
            for (i=0; i < treeNavData.length(); i++)
            {
                if (treeNavData[i].@stylegroup.toString() == "T")
                {
                    styleinfo = treeNavData[i];
                    break;
                }
            }
            
            colStyleList = [{label:"None", data:""}];
            
            if (styleinfo.children().length() != mgrid.gridCellStyle.length)
            {
                if (styleinfo.hasComplexContent())
                {
                    while (!styleinfo.hasSimpleContent())
                        delete styleinfo.children()[0];
                }
                
                /*
                for (i=0; i < mgrid.gridCellStyle.length; i++)
                {
                    var m_style:XML = new XML(<node/>);
                    var cs:CellStyle = mgrid.gridCellStyle[i];
                    m_style.@style = cs.stylename;
                    m_style.@label = cs.stylename;
                    styleinfo.appendChild(m_style);
                    
                    colStyleList.push({label:cs.stylename, data:cs.stylename});
                }
                */
                
                for each (var cs:CellStyle in mgrid.gridCellStyle)
                {
                    var m_style:XML = new XML(<node/>);
                    m_style.@style = cs.stylename;
                    m_style.@label = cs.stylename;
                    styleinfo.appendChild(m_style);
                    
                    colStyleList.push({label:cs.stylename, data:cs.stylename});
                }
            }
            
            colStyles = new ArrayCollection(colStyleList);
        }

        private function validateStyleProperty(stylename:String):void
        {
            switch (stylename)
            {
            case "normal":
            case "fixedrow":
            case "fixedcolumn":
            case "spreadsheet":
            case "subtotal":
            case "filter":
                btnRemoveStyle.includeInLayout = false;
                btnRemoveStyle.visible = false;
                break;
            default:
                btnRemoveStyle.includeInLayout = true;
                btnRemoveStyle.visible = true;
                break;
            }
            var style:CellStyle = mgrid.getStyleByName(stylename);
            style_name.label = style.stylename;

            style_backColor1.selectedColor = style.backColor1;
            style_backColor2.selectedColor = style.backColor2;
            style_backAlpha1.value = style.backAlpha1;
            style_backAlpha2.value = style.backAlpha2;
            style_borderColor.selectedColor = style.borderColor;
            style_borderThickness.value = style.borderThickness;
            style_borderThicknessTop.value = style.borderThicknessTop;
            style_borderThicknessBottom.value = style.borderThicknessBottom;
            style_borderThicknessLeft.value = style.borderThicknessLeft;
            style_borderThicknessRight.value = style.borderThicknessRight;
            
            if (style.borderThickness != style.borderThicknessTop || style.borderThickness != style.borderThicknessBottom || style.borderThickness != style.borderThicknessLeft || style.borderThickness != style.borderThicknessRight)
            {
                this.style_borderThicknessOther.selected = true;
            }
            else
            {
                this.style_borderThicknessOther.selected = false;
            }
            
            style_forecolor.selectedColor = style.foreColor;
            style_fontsize.value = style.fontSize;
            
            style_bold.selected = false;
            style_italic.selected = false;
            style_underline.selected = false;

            if (style.FontStyle & FontStyleEnum.BOLD)
                style_bold.selected = true;

        private function validateColumnProperty(col:int):void
        {
            col_name.text = mgrid.Cols.item(col).name;
            if (col_name.text == "")
                col_name.text = "col_" + col;
            col_visible.selected = mgrid.Cols.item(col).visible;
            col_editable.selected = mgrid.Cols.item(col).editable;
            col_merge.selected = mgrid.Cols.item(col).merge;
            col_showfilter.selected = mgrid.Cols.item(col).enableFilter;
            
            if (mgrid.Cols.item(col).dataField)
                col_datafield.text = mgrid.Cols.item(col).dataField;
            else
                col_datafield.text = "";
            
            col_autowidth.selected = mgrid.Cols.item(col).autowidth;
            
            if (col_autowidth.selected == false)
            {
                col_width.value = mgrid.Cols.item(col).width;
            }
            
            var i:int;
            var header:String = "";
            
            if (mgrid.Rows.fixed > 0)
            {
                for (i=0; i < mgrid.Rows.fixed; i++)
                {
                    var dt:Object;
                    
                    dt = mgrid.get_cell(CellOp.TEXT, i, col);
                    
                    if (i == 0)
                    {
                        header = (dt == null) ? "" : dt.toString();
                    }
                    else
                    {
                        header += ";" + ((dt == null) ? "" : dt.toString());
                    }
                }
            }
            
            col_header.text = header;
            col_datatype.selectedIndex = 0;
            
            for (i=0; i < datatypes.length; i++)
            {
                if (datatypes[i].data == mgrid.Cols.item(col).datatype)
                {
                    col_datatype.selectedIndex = i;
                    break;
                }
            }
            
            for (i=0; i < textalignment.length; i++)
            {
                if (textalignment[i].data == mgrid.Cols.item(col).textalign)
                {
                    col_textalign.selectedIndex = i;
                    break;
                }
            }
            
            for (i=0; i < textalignment.length; i++)
            {
                if (textalignment[i].data == mgrid.Cols.item(col).textalign_fixed)
                {
                    col_textalign_fixed.selectedIndex = i;
                    break;
                }
            }
            
            col_style.selectedIndex = 0;
            
            for (i=0; i < this.colStyleList.length; i++)
            {
                if (this.colStyleList[i].data == mgrid.Cols.item(col).style)
                {
                    col_style.selectedIndex = i;
                    break;
                }
            }
        }
