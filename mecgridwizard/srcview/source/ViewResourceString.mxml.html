<?xml version="1.0" encoding="utf-8"?>
<mx:TitleWindow xmlns:mx="http://www.adobe.com/2006/mxml" layout="absolute" width="600" height="400">
<mx:Script>
    <![CDATA[
        import mx.events.CloseEvent;
        import mx.controls.Alert;
        
        public function set ResourceXML(value:String):void
        {
            callLater(loadResourceText, [value]);
        }
        
        public function get ResourceXML():String
        {
            return resource_xml.text;
        }
        
        private function loadResourceText(txml:String):void
        {
            resource_xml.text = txml;
        }
        
        private function closeNavWindow():void
        {
            try
            {
                var sxml:XML = new XML(resource_xml.text);
            }
            catch (e:Error)
            {
                Alert.show(e.message);
                return;
            }
            
            var notify:CloseEvent = new CloseEvent(CloseEvent.CLOSE);
            dispatchEvent(notify);
        }
    ]]>
</mx:Script>
    <mx:Button label="OK" horizontalCenter="0" bottom="10" click="closeNavWindow()"/>
    <mx:TextArea id="resource_xml" left="10" right="10" top="10" bottom="40"/>
</mx:TitleWindow>
