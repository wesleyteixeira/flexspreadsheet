<?xml version="1.0" encoding="utf-8"?>
<mx:TitleWindow xmlns:mx="http://www.adobe.com/2006/mxml" layout="absolute" 
    width="400" height="136" title="New Cell Style" creationComplete="initApp()">
<mx:Script>
    <![CDATA[
        import mx.events.CloseEvent;
        import mx.controls.Alert;
        import mx.managers.PopUpManager;
        
        private function initApp():void
        {
            txtStyleName.setFocus();
        }
        
        private function closeNavWindow(cmd:String):void
        {
            PopUpManager.removePopUp(this);
            
            var notify:CloseEvent = new CloseEvent(CloseEvent.CLOSE);
            
            switch (cmd)
            {
            case "confirm":
                if (txtStyleName.text == "")
                {
                    Alert.show("Please, fill new style name to be used.");
                    return;
                }
                notify.detail = Alert.OK;
                break;
            case "cancel":
                notify.detail = Alert.CANCEL;
                break;
            }
            
            dispatchEvent(notify);
        }
    ]]>
</mx:Script>
    <mx:Label x="10" y="21" text="Style name"/>
    <mx:TextInput x="86" y="19" width="244" id="txtStyleName"/>
    <mx:Button x="138" y="64" label="OK" click="closeNavWindow('confirm')"/>
    <mx:Button x="189" y="64" label="Cancel" click="closeNavWindow('cancel')"/>
</mx:TitleWindow>
